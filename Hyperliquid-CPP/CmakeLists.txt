cmake_minimum_required(VERSION 3.20)

project(
	HyperliquidCPP
	VERSION 1.0.0
	DESCRIPTION "HYPERLIQUID C++ SDK"
	LANGUAGES CXX
)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)


#find all dependencies
find_package(boost_beast CONFIG REQUIRED)
find_package(OpenSSL REQUIRED)
find_package(nlohmann_json CONFIG REQUIRED)
find_package(msgpack-cxx CONFIG REQUIRED)
find_package(cryptopp CONFIG REQUIRED)
find_package(libsecp256k1 CONFIG REQUIRED)


#compile all .cpp which create a .HyperliquidCPP.lib
add_library(HyperliquidCPP
	src/HttpClient.cpp
	src/HyperliquidClient.cpp
	src/HyperliquidInfo.cpp
	src/HyperliquidSigner.cpp
	src/Orderbook.cpp
	src/WSConnect.cpp
	src/TradeFeed.cpp
	src/Dispatcher.cpp
	src/BBOFeed.cpp
	src/L2BookFeed.cpp
	)


add_library(HyperliquidCPP::HyperliquidCPP ALIAS HyperliquidCPP)


#set a public files that will be installed later
target_sources(HyperliquidCPP
    PUBLIC
        FILE_SET HEADERS
        BASE_DIRS include
        FILES
            include/HttpClient.h
            include/HyperliquidClient.h
            include/HyperliquidInfo.h
            include/HyperliquidSigner.h
			include/HyperliquidTypes.h
            include/Orderbook.h
            include/WSConnect.h
			include/Dispatcher.h
			include/BBOFeed.h
			include/TradeFeed.h
			include/L2BookFeed.h)
			


#tell compilers where to look development/ installed sdk
target_include_directories(HyperliquidCPP
    PUBLIC
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<INSTALL_INTERFACE:include>
)

#this will create a HLcpp.exe
add_executable(HLcpp
	test/main.cpp
 )


# link the dependencies to HYPERLIQUIDCPP
target_link_libraries(HyperliquidCPP
	PUBLIC
		OpenSSL::SSL
		OpenSSL::Crypto
		Boost::beast
		nlohmann_json::nlohmann_json
		msgpack-cxx
		cryptopp::cryptopp
		libsecp256k1::secp256k1
)


#THIS LINK HLcpp uses HyperliquidCPP
target_link_libraries(HLcpp
	PRIVATE
		HyperliquidCPP
)



#user -> run cmake --install -> HyperliquidCPP.lib -> install/, include/, lib/
install(
	TARGETS HyperliquidCPP
	EXPORT HyperliquidCPPTargets

	ARCHIVE DESTINATION lib
	LIBRARY	DESTINATION lib
	RUNTIME DESTINATION bin

	FILE_SET HEADERS

)

#Export HyperliqudCPPTargets -> HyperliqudCPPTargets.cmake(how to use the SDK)
install(
	EXPORT HyperliquidCPPTargets
	FILE HyperliquidCPPTargets.cmake
	NAMESPACE HyperliquidCPP::
	DESTINATION lib/cmake/HyperliquidCPP
)


include(CMakePackageConfigHelpers)

#create 3 .cmake at lib/cmake/HyperliquidCPP, this tell how to use the SDK
configure_package_config_file(
    cmake/HyperliquidCPPConfig.cmake.in
    ${CMAKE_CURRENT_BINARY_DIR}/HyperliquidCPPConfig.cmake
    INSTALL_DESTINATION lib/cmake/HyperliquidCPP
)

#cre
write_basic_package_version_file(
    ${CMAKE_CURRENT_BINARY_DIR}/HyperliquidCPPConfigVersion.cmake
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)

install(
    FILES
        ${CMAKE_CURRENT_BINARY_DIR}/HyperliquidCPPConfig.cmake
        ${CMAKE_CURRENT_BINARY_DIR}/HyperliquidCPPConfigVersion.cmake
    DESTINATION
        lib/cmake/HyperliquidCPP
)
